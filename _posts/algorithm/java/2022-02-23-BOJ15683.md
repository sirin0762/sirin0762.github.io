# 문제
[백준 감시 문제 링크](https://www.acmicpc.net/problem/15683)

# 내 생각
CCTV 방향에 대한 순열을 dfs로, 해당 방향을 통해 감시하는 것을 bfs로 구현해야하는 문제이다.   
백준에서 지원하는 Java 버전이 11이라 pattern matching switch를 몰라서 틀리고 있다(차후 수정).
# 코드
```java
package boj_15683;

import java.util.ArrayList;
import java.util.LinkedList;
import java.util.List;
import java.util.Queue;
import java.util.Scanner;

class CCTV {

    int number;

    int x;

    int y;

    public CCTV(int number, int x, int y) {
        this.number = number;
        this.x = x;
        this.y = y;
    }

}

public class Main {

    public static int N, M;

    public static int[][] map;

    public static int[][] copyMap;

    public static int[] output;

    public static List<CCTV> cctvList;

    public static int[] dx = {-1, 0, 1, 0};

    public static int[] dy = {0, 1, 0, -1};

    public static int answer = Integer.MAX_VALUE;

    public static boolean[][] visited;

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        N = sc.nextInt();
        M = sc.nextInt();
        map = new int[N][M];
        cctvList = new ArrayList<>();

        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                map[i][j] = sc.nextInt();
                if (map[i][j] != 0 && map[i][j] != 6) {
                    cctvList.add(new CCTV(map[i][j], i, j));
                }
            }
        }
        output = new int[cctvList.size()];
        permutation(0, cctvList.size());
        System.out.println(answer);
    }

    private static void permutation(int depth, int r) {
        if (depth == r) {
            copyMap = new int[N][M];
            for (int i = 0; i < map.length; i++) {
                System.arraycopy(map[i], 0, copyMap[i], 0, map[i].length);
            }
            for (int i = 0; i < cctvList.size(); i++) {
                direction(cctvList.get(i), output[i]);
            }

            getBlindSpot();
            return;
        }
        for (int i = 0; i < 4; i++) {
            output[depth] = i;
            permutation(depth + 1, r);
        }
    }

    private static void direction(CCTV cctv, int d) {
        int cctvNum = cctv.number;

        switch (cctvNum) {
            case 1 -> {
                watch(cctv, d);
            }
            case 2 -> {
                switch (d) {
                    case 0, 2 -> {
                        watch(cctv, 0);
                        watch(cctv, 2);
                    }
                    case 1, 3 -> {
                        watch(cctv, 1);
                        watch(cctv, 3);
                    }
                }
            }
            case 3 -> {
                switch (d) {
                    case 0 -> {
                        watch(cctv, 0);
                        watch(cctv, 1);
                    }
                    case 1 -> {
                        watch(cctv, 1);
                        watch(cctv, 2);
                    }
                    case 2 -> {
                        watch(cctv, 2);
                        watch(cctv, 3);
                    }
                    case 3 -> {
                        watch(cctv, 3);
                        watch(cctv, 0);
                    }
                }
            }
            case 4 -> {
                switch (d) {
                    case 0 -> {
                        watch(cctv, 0);
                        watch(cctv, 1);
                        watch(cctv, 3);
                    }
                    case 1 -> {
                        watch(cctv, 0);
                        watch(cctv, 1);
                        watch(cctv, 2);
                    }
                    case 2 -> {
                        watch(cctv, 1);
                        watch(cctv, 2);
                        watch(cctv, 3);
                    }
                    case 3 -> {
                        watch(cctv, 0);
                        watch(cctv, 2);
                        watch(cctv, 3);
                    }
                }
            }
            default -> {
                watch(cctv, 0);
                watch(cctv, 1);
                watch(cctv, 2);
                watch(cctv, 3);
            }
        }
    }

    private static void watch(CCTV cctv, int d) {
        Queue<CCTV> queue = new LinkedList<>();
        visited = new boolean[N][M];

        queue.add(cctv);
        visited[cctv.x][cctv.y] = true;

        while(!queue.isEmpty()) {
            CCTV now = queue.poll();
            int nx = now.x + dx[d];
            int ny = now.y + dy[d];

            if(nx < 0 || nx >= N || ny < 0 || ny >= M || copyMap[nx][ny] == 6) break;


            if(copyMap[nx][ny] == 0) {
                copyMap[nx][ny] = -1; // 빈칸이라면 감시할 수 있다는 의미로 -1
                queue.add(new CCTV(cctv.number, nx, ny));
            } else { // 다른 cctv가 있거나 이미 감시된 칸이라면
                queue.add(new CCTV(cctv.number, nx, ny)); // 그냥 통과
            }
        }
    }

    public static void getBlindSpot() {
        int cnt = 0;
        for(int i = 0; i < N; i++) {
            for(int j = 0; j < M; j++) {
                if(copyMap[i][j] == 0) {
                    cnt++;
                }
            }
        }
        answer = Math.min(answer, cnt);
    }

}

```

다음은 비트마스킹을 통한 풀이이다.
1. add는 x를 추가하기 때문에 `1 << x` 와 `OR` 연산을 활용해 추가한다.
2. remove는 x를 제거하기 때문에 `1 << x` 를 `NOT` 연산을 사용해 비트를 뒤집어 `AND` 연산 해준다.
3. check는 x의 체크여부를 확인하기 위해 `1 << x` 와 마스킹한 값을 `AND` 연산하여 0이 아닌지 확인한다.
4. toggle은 연산은 x의 연산을 뒤집어야 하므로 `1 << x` 와 `XOR` 연산을 사용한다.
5. all은 보든 비트를 체크하기 때문에 (1 << 21) - 1로 값을 변경해준다.
6. empty는 공집합 이기 때문에 0으로 값을 변경한다.
```java
package boj_11723;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class Main2 {

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int M = Integer.parseInt(br.readLine());

        int s = 0;
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < M; i++) {
            String[] input = br.readLine().split(" ");

            switch (input[0]) {
                case "all":
                    s = (1 << 21) - 1;
                    break;
                case "empty":
                    s = 0;
                    break;
                default:
                    int x = Integer.parseInt(input[1]);
                    switch (input[0]) {
                        case "add":
                            s |= (1 << x);
                            break;
                        case "remove":
                            s &= ~(1 << x);
                            break;
                        case "check":
                            sb.append((s & (1 << x)) != 0 ? 1 : 0).append('\n');
                            break;
                        case "toggle":
                            s ^= (1 << x);
                            break;
                    }
            }
        }
        System.out.println(sb);
    }
}
```
